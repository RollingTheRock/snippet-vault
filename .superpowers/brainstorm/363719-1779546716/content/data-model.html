数据库表结构

SQLite Schema
-- snippets 表（代码片段）
CREATE TABLE snippets (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  title TEXT NOT NULL,          -- 片段标题
  content TEXT NOT NULL,       -- 代码内容
  language TEXT NOT NULL,      -- 语言类型: html/css/js/python/java...
  description TEXT,             -- 描述说明
  created_at INTEGER,           -- 创建时间戳
  updated_at INTEGER,           -- 更新时间戳
  copy_count INTEGER DEFAULT 0 -- 被复制次数（可选统计）
);

CREATE INDEX idx_snippets_title ON snippets(title);
CREATE INDEX idx_snippets_language ON snippets(language);
-- tags 表（标签）
CREATE TABLE tags (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  name TEXT NOT NULL UNIQUE, -- 标签名
  color TEXT                         -- 标签颜色 (#4a9eff)
  created_at INTEGER
);
-- snippet_tags 表（多对多关联）
CREATE TABLE snippet_tags (
  snippet_id INTEGER NOT NULL,
  tag_id INTEGER NOT NULL,
  PRIMARY KEY (snippet_id, tag_id),
  FOREIGN KEY (snippet_id) REFERENCES snippets(id) ON DELETE CASCADE,
  FOREIGN KEY (tag_id) REFERENCES tags(id) ON DELETE CASCADE
);

数据流转示意

快启搜索框 → 输入关键词 → IPC → 主进程 → SQLite 模糊查询 → 返回结果列表
↓ Enter（复制）
快启搜索框 → IPC → 主进程 → 写入剪贴板 → 隐藏窗口
↓ Cmd+Enter（预览）
快启搜索框 → IPC → 主进程 → 创建预览窗口 → 注入代码内容
↓ 主窗口 CRUD
主窗口 → IPC → 主进程 → SQLite 增删改查 → 同步更新前端状态
