function debounce(fn, delay) {
  let timer = null;
  return (...args) => {
    clearTimeout(timer);
    timer = setTimeout(() => fn(...args), delay);
  };
}

托盘与快捷键

  • 🖥️ 系统托盘驻留：右键菜单 → 打开管理窗口 / 快启搜索 / 退出
  • ⌨️ Alt+Space（可配置）：全局呼出/隐藏快启搜索框
  • ⌨️ Esc：关闭快启搜索框（失焦自动隐藏）
  • ⌨️ Cmd/Ctrl+N：在主窗口中新建片段
  • ⌨️ Cmd/Ctrl+S：保存当前编辑的片段
